tree("Root")
	fallback
		tree "Idle"
		tree "GetBolts"
		tree "GoReload"

tree "Idle"
	while not IsAware
		Idle

tree "GetBolts"
	sequence
		not AsBolts
		ChooseRandomHouse
		GetBolts
		Wait 0.1
		AsArrived
		tree "GettingBolts"

tree "GettingBolts"
	while not IsVillagerFullOfBolts
		repeat
			sequence
				IsAware
				Wait .5
				GettingBolts

tree "GoReload"
	sequence
		IsAware
		ChooseRandomBalista
		GoReload
		Wait 0.1
		AsArrived
		tree "Reloading"

tree "Reloading"
	sequence
		IsAware
		while AsBolts
			repeat
				sequence
					Wait .5
					Reloading
